ROLES Y USUARIOS.

--CREAR USUARIO Y EL ROL
CREATE ROLE administrador WITH LOGIN PASSWORD 'acces123' VALID UNTIL '2024-12-31 23:59:59'

CREATE ROLE "adminVeterinaria" WITH
	LOGIN
	SUPERUSER
	CREATEDB
	CREATEROLE
	INHERIT
	NOREPLICATION
	BYPASSRLS
	CONNECTION LIMIT -1
	VALID UNTIL '2024-12-31T23:55:00-05:00' 
	PASSWORD 'xxxxxx';
COMMENT ON ROLE "adminVeterinaria" IS 'Usuario para administrar la DB';

CREATE ROLE usuario1 WITH
	LOGIN
	NOSUPERUSER
	NOCREATEDB
	NOCREATEROLE
	INHERIT
	NOREPLICATION
	NOBYPASSRLS
	CONNECTION LIMIT -1
	VALID UNTIL '2024-11-30T12:05:00-05:00' 
	PASSWORD 'xxxxxx';


--comandos para dar permisos a usuario regular
psql -U postgres -d usuario1
GRANT CONNECT ON DATABASE db_clinica_veterinaria TO usuario1;
GRANT SELECT ON ALL TABLES IN SCHEMA esquemas(1) TO usuario1;
